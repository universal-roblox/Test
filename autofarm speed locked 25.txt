speedBtn.MouseButton1Click:Connect(function()
    flySpeed = flySpeed + 0
    if flySpeed > 25 then flySpeed = 25 end
    speedBtn.Text = "Speed: " .. flySpeed
end)

local function flyTo(pos, speed)
    if not rootPart then return end
    local distance = (pos - rootPart.Position).Magnitude
    local duration = distance / speed
    local tweenInfo = TweenInfo.new(duration, Enum.EasingStyle.Linear)
    local goal = {CFrame = CFrame.new(pos)}
    local tween = TweenService:Create(rootPart, tweenInfo, goal)
    tween:Play()
    tween.Completed:Wait()
end

toggleBtn.MouseButton1Click:Connect(function()
    isActive = not isActive
    toggleBtn.Text = isActive and "Auto Farm: ACTIVE" or "Auto Farm: OFF"
    toggleBtn.BackgroundColor3 = isActive and Color3.fromRGB(220, 20, 60) or Color3.fromRGB(255, 182, 193)

    if isActive then  
        startTime = tick()  
        
        task.spawn(function()  
            while isActive do  
                local elapsed = tick() - startTime  
                timerLabel.Text = "Time: " .. math.floor(elapsed) .. "s"  
                local rate = elapsed > 0 and math.floor((collected / elapsed) * 3600) or 0  
                rateLabel.Text = "Coins/Hr: " .. rate  
                task.wait(0.5)  
            end  
        end)  

        task.spawn(function()  
            while isActive do  
                character = player.Character or player.CharacterAdded:Wait()  
                rootPart = character:FindFirstChild("HumanoidRootPart")  
                if rootPart then  
                    local closest, shortest = nil, math.huge  
                    for _, obj in ipairs(workspace:GetDescendants()) do  
                        -- Targeting both standard coins and event coins
                        if obj:IsA("BasePart") and (obj.Name == "Coin_Server" or obj.Name == "ValentineCoin") then  
                            local dist = (obj.Position - rootPart.Position).Magnitude  
                            if dist < shortest and dist < 300 and not visitedPositions[obj] then  
                                closest = obj  
                                shortest = dist  
                            end  
                        end  
                    end  

                    if closest and closest.Parent then  
                        flyTo(closest.Position, flySpeed)  
                        visitedPositions[closest] = true  
                        collected = collected + 1  
                        collectSound:Play()  
                        counterLabel.Text = "Collected: " .. collected  
                    end  
                end  
                task.wait(0.1)  
            end  
        end)  
    end
end)
